    <script>
        function initMap() {
            var service = new google.maps.places.PlacesService(document.createElement('div'));
            
            // O Place ID do seu negócio (você pode obter o ID do Google Maps)
            var placeId = 'ChIJzV1fKoO5zJQR2omaxk6Aga4';  // Substitua com o Place ID real
            
            service.getDetails({
                placeId: placeId
            }, function(place, status) {
                if (status === google.maps.places.PlacesServiceStatus.OK) {
                    var reviewsContainer = document.getElementById('avaliacoes');
                    
                    place.reviews.forEach(function(review) {
                        var reviewCard = document.createElement('div');
                        reviewCard.className = 'avaliacao-card';
                        
                        // Foto do avaliador (se disponível)
                        var photo = review.profile_photo_url ? `<img src="${review.profile_photo_url}" alt="Foto de ${review.author_name}">` : '';
                        
                        reviewCard.innerHTML = `
                            <div class="foto">${photo}</div>
                            <div class="conteudo">
                                <p class="nome">${review.author_name}</p>
                                <p class="texto">${review.text}</p>
                                <p class="stars">Avaliação: ${review.rating} estrelas</p>
                            </div>
                        `;
                        
                        reviewsContainer.appendChild(reviewCard);
                    });
                }
            });
        }

        // Inicializa o mapa e pega as avaliações
        window.onload = initMap;
    </script>
